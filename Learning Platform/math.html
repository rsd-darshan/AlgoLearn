Limits

// Example: Calculating a limit
function limit(f, a, delta) {
    return (f(a + delta) - f(a)) / delta;
}

    return x * x;
}

Derivatives

A derivative represents an instantaneous rate of change. It is a fundamental tool of calculus.

// Example: Calculating the derivative of f(x) = x^2
function derivative(f, x, delta) {
    return (f(x + delta) - f(x)) / delta;
}

function f(x) {
    return x * x;
}

let delta = 0.0001;

Integrals

An integral is the reverse process of differentiation. It is used to find areas, volumes, central points, and many useful things.

// Example: Calculating the integral of f(x) = x^2
function integral(f, a, b, n) {
    let sum = 0;
    let delta = (b - a) / n;
    for (let i = 0; i < n; i++) {
        let x = a + i * delta;
        sum += f(x) * delta;
    }
    return sum;
}

function f(x) {
    return x * x;
}

let a = 0;
let b = 2;
let n = 1000;
let integ = integral(f, a, b, n);
console.log(integ);  // Output: 2.666666666666667

Greatest Common Divisor (GCD)

The greatest common divisor of two integers is the largest positive integer that divides each of the integers without a remainder.

// Example: Calculating GCD using Euclidean algorithm
                        function gcd(a, b) {
                            while (b !== 0) {
                                let temp = b;
                                b = a % b;
                                a = temp;
                            }
                            return a;
                        }
                        
                        console.log(gcd(48, 18));  // Output: 6

Basic Concepts

Group theory studies the algebraic structures known as groups, which are sets equipped with an operation that combines any two elements to form a third element.

// Example: Group properties
                        // A group (G, *) must satisfy four conditions: closure, associativity, identity, and invertibility
                        function isGroup(G, operation) {
                            // Check closure
                            for (let a of G) {
                                for (let b of G) {
                                    if (!G.includes(operation(a, b))) return false;
                                }
                            }
                            // Check associativity
                            for (let a of G) {
                                for (let b of G) {
                                    for (let c of G) {
                                        if (operation(a, operation(b, c)) !== operation(operation(a, b), c)) return false;
                                    }
                                }
                            }
                            // Check identity
                            let identity = G.find(e => G.every(a => operation(a, e) === a && operation(e, a) === a));
                            if (identity === undefined) return false;
                            // Check invertibility
                            for (let a of G) {
                                if (!G.find(b => operation(a, b) === identity && operation(b, a) === identity)) return false;
                            }
                            return true;
                        }
                        
                        let G = [0, 1, 2];
                        let operation = (a, b) => (a + b) % 3;
                        console.log(isGroup(G, operation));  // Output: true

// Example: Checking for subgroups

Introduction to Differential Equations

Differential equations are equations that involve an unknown function and its derivatives. They describe various exponential growths and decays, oscillations, and many other natural phenomena.

// Example: Solving a simple differential equation
    // dy/dx = y
    function solveDifferentialEquation(y0, x0, x, h) {
        let y = y0;
        while (x0 < x) {
            y += h * y;
            x0 += h;
        }
        return y;
    }
    
    let y0 = 1;
    let x0 = 0;
    let x = 1;
    let h = 0.01;
    console.log(solveDifferentialEquation(y0, x0, x, h));  // Output: Approximation of e^1

Separable Equations

A separable differential equation is one that can be separated into a product of functions of each variable.

// Example: Solving a separable differential equation
    // dy/dx = g(x) * h(y)
    function solveSeparableEquation(g, h, y0, x0, x, h) {
        let y = y0;
        while (x0 < x) {
            y += h * g(x0) * h(y);
            x0 += h;
        }
        return y;
    }
    
    let g = (x) => x;
    let h = (y) => y;
    console.log(solveSeparableEquation(g, h, y0, x0, x, h));  // Output: Approximation of e^(x^2/2)

Linear Differential Equations

Linear differential equations are of the form dy/dx + P(x)y = Q(x).

// Example: Solving a linear differential equation
    // dy/dx + P(x)y = Q(x)
    function solveLinearEquation(P, Q, y0, x0, x, h) {
        let y = y0;
        while (x0 < x) {
            y += h * (Q(x0) - P(x0) * y);
            x0 += h;
        }
        return y;
    }
    
    let P = (x) => -2;
    let Q = (x) => 3;
    console.log(solveLinearEquation(P, Q, y0, x0, x, h));  // Output: Approximation of the solution

Root Finding

Numerical methods are techniques to approximate mathematical procedures. Root finding algorithms are used to find zeros of functions.

// Example: Using the bisection method
    function bisection(f, a, b, tol) {
        let c;
        while ((b - a) / 2 > tol) {
            c = (a + b) / 2;
            if (f(c) === 0) return c;
            else if (f(c) * f(a) < 0) b = c;
            else a = c;
        }
        return c;
    }
    
    let f = (x) => x * x - 2;

Numerical Integration

Numerical integration algorithms approximate the integral of a function.

// Example: Using the trapezoidal rule
    function trapezoidal(f, a, b, n) {
        let h = (b - a) / n;
        let sum = (f(a) + f(b)) / 2;
        for (let i = 1; i < n; i++) {
            sum += f(a + i * h);
        }
        return sum * h;
    }
    
    let f = (x) => x * x;
    console.log(trapezoidal(f, 0, 1, 100));  // Output: Approximation of integral of x^2 from 0 to 1

Groups

In mathematics, a group is a set equipped with a binary operation that satisfies certain axioms.

// Example: Group properties
    // A group (G, *) must satisfy four conditions: closure, associativity, identity, and invertibility
    function isGroup(G, operation) {
        // Check closure
        for (let a of G) {
            for (let b of G) {
                if (!G.includes(operation(a, b))) return false;
            }
        }
        // Check associativity
        for (let a of G) {
            for (let b of G) {
                for (let c of G) {
                    if (operation(a, operation(b, c)) !== operation(operation(a, b), c)) return false;
                }
            }
        }
        // Check identity
        let identity = G.find(e => G.every(a => operation(a, e) === a && operation(e, a) === a));
        if (identity === undefined) return false;
        // Check invertibility
        for (let a of G) {
            if (!G.find(b => operation(a, b) === identity && operation(b, a) === identity)) return false;
        }
        return true;
    }
    
    let G = [0, 1, 2];
    let operation = (a, b) => (a + b) % 3;
    console.log(isGroup(G, operation));  // Output: true

Rings

A ring is an algebraic structure consisting of a set equipped with two binary operations satisfying properties analogous to those of addition and multiplication.

// Example: Ring properties
    // A ring (R, +, *) must satisfy the ring axioms
    function isRing(R, add, multiply) {
        // Check if (R, +) is an abelian group
        if (!isGroup(R, add)) return false;
        // Check if * is associative
        for (let a of R) {
            for (let b of R) {
                for (let c of R) {
                    if (multiply(a, multiply(b, c)) !== multiply(multiply(a, b), c)) return false;
                }
            }
        }
        // Check distributive laws
        for (let a of R) {
            for (let b of R) {
                for (let c of R) {
                    if (multiply(a, add(b, c)) !== add(multiply(a, b), multiply(a, c))) return false;
                    if (multiply(add(a, b), c) !== add(multiply(a, c), multiply(b, c))) return false;
                }
            }
        }
        return true;
    }
    
    let R = [0, 1, 2];
    let add = (a, b) => (a + b) % 3;
    let multiply = (a, b) => (a * b) % 3;
    console.log(isRing(R, add, multiply));  // Output: true

Fields

A field is an algebraic structure in which addition, subtraction, multiplication, and division are defined and satisfy certain properties.

// Example: Field properties
    // A field (F, +, *) must satisfy the field axioms
    function isField(F, add, multiply) {
        // Check if (F, +) is an abelian group
        if (!isGroup(F, add)) return false;
        // Check if (F \ {0}, *) is an abelian group
        let FNonZero = F.filter(x => x !== 0);
        if (!isGroup(FNonZero, multiply)) return false;
        // Check distributive laws
        for (let a of F) {
            for (let b of F) {
                for (let c of F) {
                    if (multiply(a, add(b, c)) !== add(multiply(a, b), multiply(a, c))) return false;
                }
            }
        }
        return true;
    }
    
    let F = [0, 1, 2];
    console.log(isField(F, add, multiply));  // Output: true

Chinese Remainder Theorem

The Chinese Remainder Theorem is a result about congruences in number theory that allows us to solve systems of simultaneous linear congruences with different moduli.

// Example: Applying the Chinese Remainder Theorem
function chineseRemainder(n, a) {
let sum = 0;
let prod = n.reduce((a, b) => a * b);

for (let i = 0; i < n.length; i++) {
let p = prod / n[i];
sum += a[i] * mulInv(p, n[i]) * p;
}

return sum % prod;
}

function mulInv(a, b) {
let b0 = b;
let x0 = 0;
let x1 = 1;

if (b === 1) return 1;

while (a > 1) {
let q = Math.floor(a / b);
let t = b;
b = a % b;
a = t;
t = x0;
x0 = x1 - q * x0;
x1 = t;
}

if (x1 < 0) x1 += b0;

return x1;
}

let n = [3, 5, 7];
let a = [2, 3, 2];
console.log(chineseRemainder(n, a));  // Output: 23

Introduction to Fourier Series

Fourier analysis is the study of how functions can be represented or approximated by sums of simpler trigonometric functions. A Fourier series decomposes a periodic function into a sum of sine and cosine terms.

// Example: Calculating Fourier coefficients
function fourierSeries(f, n, L) {
let a0 = (1 / L) * integral((x) => f(x), -L, L);
let series = [a0 / 2];
for (let k = 1; k <= n; k++) {
let ak = (1 / L) * integral((x) => f(x) * Math.cos((k * Math.PI * x) / L), -L, L);
let bk = (1 / L) * integral((x) => f(x) * Math.sin((k * Math.PI * x) / L), -L, L);
series.push(ak, bk);
}
return series;
}

function integral(f, a, b) {
let n = 1000;
let h = (b - a) / n;
let sum = (f(a) + f(b)) / 2;
for (let i = 1; i < n; i++) {
sum += f(a + i * h);
}
return sum * h;
}
